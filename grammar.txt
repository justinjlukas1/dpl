body            : O_BRACE statements C_BRACE

statements      : statement
                | statement statements

statement       : expression
                |	definition
                | assignment
                |	ifStatement
                | forLoop
                | whileLoop
                |	RESULT expression

expression      : unary
                | binary

binary          : operator expression optBinaryItems

optBinaryItems  : expression
                | NULL

unary           : INTEGER
                |	REAL
                | STRING
                | functionCall
                | lambda
                | list
                |	object
                | O_PAREN expression C_PAREN
                | EX_POINT unary  //! x1
                | MINUS unary
|	MINUS unary
|	list
|	O_PAREN expression C_PAREN
|	VARIABLE list //function call
|	lambda list //lambda call
//              | arrayInit

object          : VARIABLE
                | VARIABLE DOT object

functionCall    : object list

lambda          : LAMBDA procedure

procedure       : list body

list            : O_PAREN listItem C_PAREN

listItem        : expression
                | expression listItem
                | NULL

definition	:	DEF VARIABLE
|	DEF VARIABLE expression
|	DEF VARIABLE list body //function def
|	DEF VARIABLE lambda //lambda def

lambda :	LAMBDA list body

ifStatement : IF expression body optIfElse

unary :	INTEGER
|	STRING
|	BOOLEAN
|	VARIABLE
|	MINUS unary
|	list
|	O_PAREN expression C_PAREN
|	VARIABLE list //function call
|	lambda list //lambda call

operator	:	PLUS
|	MINUS
|	MULTIPLY
|	DIVIDE
|	EQUALS //=, numeric equality
|	LESS_THAN
|	GREATER_THAN
|	INDEX //#
|	AND
|	OR
|	NOT
|	EXPONENT

optIfElse	:	ELSE body
|	ELSE ifStatement
|	*empty*

list :	O_BRACE optListItems C_BRACE

optListItems:	expression
|	expression COMMA optListItems
|	*empty*
